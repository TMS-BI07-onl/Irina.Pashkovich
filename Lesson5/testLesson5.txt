
1.Выберите из таблицы Водители записи, где стаж водителя не равен 3, 5, 6, 10 и место приписки 'eee'
SELECT *
FROM Водители 
WHERE Стаж NOT IN (3,5,6,10)  AND  Место прописки = ‘eee’

2. Найдите номера паспортов водитлей которые ни разу не совершали поездки 
SELECT № паспорта
FROM Водители AS Водители 
LEFT JOIN Диспетчерская AS Диспетчерская 
ON Водители.Код водителя = Диспетчерская.Код водителя
WHERE Диспетчерская.Код водителя IS NULL

3.Написать запрос на выборку автос годом выпуска = 2015 м его маршруты. Вывести рег номер, время прибытия и отбытия, пункт назначения
SELECT Регистрационый № авто, Время прибытия, Время отбытия, Пункт назначения
FROM Автомобили AS Автомобили

JOIN Диспетчерская AS Диспетчерская 
ON Автомобили.Код авто = Диспетчерская.Код авто

JOIN Маршрут AS Маршрут
ON Диспетчерская.Путёвка = Маршрут.Код маршрута

WHERE Год выпуска авто = 2015
